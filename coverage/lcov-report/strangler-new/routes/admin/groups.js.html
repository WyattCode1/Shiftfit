function _save_credit_card (req, res, callback) {
    console.info("Saving a new credit card method for a group");
    process.env.NODE_TLS_REJECT_UNAUTHORIZED = "0"
 
    var request_url = config.url('add_credit_card_group');
    var token = req.cookies.ZintroLogin;
    console.info("Request url: " + request_url);
 
    var formOfPayment = {
        id: '0'
    }
    var month = {
        monthId: req.body.card_month,
        monthName: req.body.card_month
    }
    var year = {
        yearId: req.body.card_year,
        yearName: req.body.card_year
    }
    var paymentInfo = {
        creditCardCheck: true,
        paypalCheck: false,
        isUpgrade: false,
        isGroup: true,
        groupId: req.body.group_id,
        cvcNumber: req.body.cvc_number,
        nameOnCreditCard: req.body.name,
        creditCardNumber: req.body.card_number,
        formOfPayment: formOfPayment,
        month: month,
        year: year
    };
 
    var options = {
        url: request_url,
        method: 'POST',
        headers: {
            'app-token': token,
            Accepts: 'application/json'
        },
        json: paymentInfo
    };
 
    request(options, function (err, response, body) {
        if (err || body.briefError) {
            console.info("Failed the creation of the credit card");
            return res.status(500).send();
        } else {
            console.info("Successfully creation of the credit card");
            return callback();
        }
    });
}
